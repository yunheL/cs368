/* what are all the movie title, showtimes showings at a particular theater */
SELECT title, showtime, tid FROM showings, movies where showings.tid = 0 AND movies.mid = showings.mid;

/* What are all the movies directed by a director */
SELECT did, title FROM movies, md WHERE md.did = 0 AND movies.mid = md.mid;

/* who direct a particular movie */
